\[
\begin{matrix}
33:! & 34: " & 35:\# & \cdots & 42:* \\
43:+ & 44: , & 45:- & \cdots & 52:4 \\
\vdots & & & & \vdots \\
113:q & 114:r & 115:s & \cdots & 122:z
\end{matrix}
\]
\begin{solution}

		\footnotesize

		\label{lst:5.2}
